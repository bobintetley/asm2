###
Produces a breakdown of income and expenses over a period (profit and loss).
###
en
###
$VAR from DATE Enter from date$
$VAR to DATE Enter to date$

SELECT 
a.Code,
ABS(COALESCE((SELECT SUM(Amount) FROM accountstrx
WHERE accountstrx.DestinationAccountID = a.ID AND accountstrx.TrxDate BETWEEN '$@from$' AND '$@to$'), 0) - COALESCE((SELECT SUM(Amount) FROM accountstrx WHERE accountstrx.SourceAccountID = a.ID AND accountstrx.TrxDate BETWEEN '$@from$' AND '$@to$'), 0)) AS fig,
l.AccountType
FROM accounts a
INNER JOIN lksaccounttype l ON l.ID = a.AccountType
WHERE a.AccountType IN (2,3,4,5)
ORDER BY l.AccountType
